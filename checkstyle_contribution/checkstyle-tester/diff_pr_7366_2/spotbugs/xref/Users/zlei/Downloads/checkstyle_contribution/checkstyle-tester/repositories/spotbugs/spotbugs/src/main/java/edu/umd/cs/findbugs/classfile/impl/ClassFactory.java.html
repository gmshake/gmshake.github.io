37  /**
38   * Factory to create codebase/classpath/classfile objects.
39   *
40   * @author David Hovemeyer
41   */
42  public class ClassFactory implements IClassFactory {
43      private static IClassFactory theInstance = new ClassFactory();
44  
45      private ClassFactory() {
46      }
47  
48      public static IClassFactory instance() {
49          return theInstance;
50      }
51  
52      /*
53       * (non-Javadoc)
54       *
55       * @see edu.umd.cs.findbugs.classfile.impl.IClassFactory#createClassPath()
56       */
57      @Override
58      public IClassPath createClassPath() {
59          return new ClassPathImpl();
60      }
61  
62      /*
63       * (non-Javadoc)
64       *
65       * @see
66       * edu.umd.cs.findbugs.classfile.IClassFactory#createClassPathBuilder(edu
67       * .umd.cs.findbugs.classfile.IErrorLogger)
68       */
69      @Override
70      public IClassPathBuilder createClassPathBuilder(IErrorLogger errorLogger) {
71          return new ClassPathBuilder(this, errorLogger);
72      }
73  
74      /*
75       * (non-Javadoc)
76       *
77       * @see edu.umd.cs.findbugs.classfile.impl.IClassFactory#
78       * createFilesystemCodeBaseLocator(java.lang.String)
79       */
80      @Override
81      public ICodeBaseLocator createFilesystemCodeBaseLocator(String pathName) {
82          // Attempt to canonicalize the pathname.
83          // It's not fatal if we can't.
84          try {
85              pathName = new File(pathName).getCanonicalPath();
86          } catch (IOException e) {
87              // Ignore
88          }
89  
90          return new FilesystemCodeBaseLocator(pathName);
91      }
92  
93      /*
94       * (non-Javadoc)
95       *
96       * @see edu.umd.cs.findbugs.classfile.IClassFactory#
97       * createNestedArchiveCodeBaseLocator
98       * (edu.umd.cs.findbugs.classfile.ICodeBase, java.lang.String)
99       */
100     @Override
101     public ICodeBaseLocator createNestedArchiveCodeBaseLocator(ICodeBase parentCodeBase, String path) {
102         return new NestedZipFileCodeBaseLocator(parentCodeBase, path);
103     }
104 
105     static IScannableCodeBase createFilesystemCodeBase(FilesystemCodeBaseLocator codeBaseLocator) throws IOException {
106         String fileName = codeBaseLocator.getPathName();
107 
108         File file = new File(fileName);
109 
110         if (!file.exists() || !file.canRead()) {
111             return createEmptyCodeBase(codeBaseLocator, file);
112         } else if (file.isDirectory()) {
113             return new DirectoryCodeBase(codeBaseLocator, file);
114         } else if (!file.isFile()) {
115             return createEmptyCodeBase(codeBaseLocator, file);
116         } else if (fileName.endsWith(".class")) {
117             return new SingleFileCodeBase(codeBaseLocator, fileName);
118         } else if (fileName.endsWith(File.separator + "jrt-fs.jar")) {
119             return new JrtfsCodeBase(codeBaseLocator, fileName);
120         } else {
121             try {
122                 return ZipCodeBaseFactory.makeZipCodeBase(codeBaseLocator, file);
123             } catch (IOException e) {
124                 return createEmptyCodeBase(codeBaseLocator, file);
125             }
126         }
127     }
128 
129     private static IScannableCodeBase createEmptyCodeBase(FilesystemCodeBaseLocator codeBaseLocator, File file) {
130         if (FindBugs2.DEBUG) {
131             System.out.println("Ignoring unreadable or non-existent file " + file);
132         }
133         return new EmptyCodeBase(codeBaseLocator);
134     }
135 
136     static IScannableCodeBase createNestedZipFileCodeBase(NestedZipFileCodeBaseLocator codeBaseLocator)
137             throws ResourceNotFoundException, IOException {
138         return new NestedZipFileCodeBase(codeBaseLocator);
139     }

143      *
144      * @see
145      * edu.umd.cs.findbugs.classfile.IClassFactory#createAnalysisCache(edu.umd
146      * .cs.findbugs.classfile.IClassPath)
147      */
148     @Override
149     public IAnalysisCache createAnalysisCache(IClassPath classPath, BugReporter errorLogger) {
150         IAnalysisCache analysisCache = new AnalysisCache(classPath, errorLogger);
151         return analysisCache;
152     }
